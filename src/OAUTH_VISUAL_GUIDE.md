async function createFounderProfile(userId, email, name) {
  
  // STEP 1: Check if profile already exists
  const { data: existing } = await supabase
    .from('founder_profiles')
    .select('id')
    .eq('user_id', userId)
    .single();
  
  if (existing) {
    return { success: true }; // Already exists, skip
  }
  
  // STEP 2: Check if user is admin
  const { data: adminCheck } = await supabase
    .from('admin_users')
    .select('id')
    .eq('user_id', userId)
    .single();
  
  if (adminCheck) {
    return { success: false }; // Admin, don't create founder
  }
  
  // STEP 3: Create founder profile
  await supabase.from('founder_profiles').insert({
    user_id: userId,
    email: email,
    name: name,
    business_name: `${name} Business`,
    current_stage: 1,
    current_week: 1,
    account_status: 'active',
    subscription_status: 'trial'
  });
  
  // STEP 4: Initialize stage progress
  for (let stage = 1; stage <= 5; stage++) {
    await supabase.from('stage_progress').insert({
      user_id: userId,
      stage_number: stage,
      status: stage === 1 ? 'in_progress' : 'locked'
    });
  }
  
  return { success: true };
}
